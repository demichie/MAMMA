733  + sum( alfa_d_1(1:n_gas) * k_d(1:n_gas) )
734 
735  c2_1 = 1.d0 / ( k_1 * rho_1 )
736

741  rho_g(1:n_gas) )**2.0 - dcmplx(2.0,0.d0) * a_g(1:n_gas) * rho_g(1:n_gas)
